// PROXY CORS
// On essaie plusieurs proxies publics dans l'ordre
// Si l'un echoue, on passe au suivant automatiquement
// ============================================================
const PROXIES = [
  'https://corsproxy.io/?',
  'https://api.allorigins.win/raw?url=',
  'https://cors-anywhere.herokuapp.com/',
];
let proxyIdx = 0;

function proxyUrl(url) {
  const p = PROXIES[proxyIdx];
  if (p === 'https://corsproxy.io/?') return p + encodeURIComponent(url);
  if (p === 'https://api.allorigins.win/raw?url=') return p + encodeURIComponent(url);
  return p + url;
}

async function pfetch(url, opts={}) {
  // Essayer d'abord sans proxy (meme domaine ou si cookies necessaires)
  // Puis avec proxy si echec
  let lastErr;
  for (let i = 0; i < PROXIES.length; i++) {
    proxyIdx = i;
    try {
      const purl = proxyUrl(url);
      const headers = Object.assign({
        'X-Requested-With': 'XMLHttpRequest',
      }, opts.headers||{});
      const r = await fetch(purl, Object.assign({}, opts, {
        headers,
        credentials: 'omit', // proxy ne peut pas transmettre les cookies
      }));
      if (r.ok || r.status < 500) {
        log('Proxy OK: ' + PROXIES[i].split('/')[2], 'ld');
        return r;
      }
    } catch(e) {
      lastErr = e;
      log('Proxy ' + PROXIES[i].split('/')[2] + ' echoue', 'lw');
    }
  }
  throw lastErr || new Error('Tous les proxies ont echoue');
}

// ============================================================
// NOTE IMPORTANTE SUR LES COOKIES :
// Les proxies CORS ne transmettent pas les cookies de session.
// On va donc passer les cookies manuellement dans les headers.
// Apres le login, on recupere le PHPSESSID depuis la reponse
// et on l'injecte dans toutes les requetes suivantes.
// ============================================================
let SESSION_COOKIE = '';

async function pfetchWithSession(url, opts={}) {
  const headers = Object.assign({}, opts.headers||{});
  if (SESSION_COOKIE) headers['Cookie'] = SESSION_COOKIE;
  return pfetch(url, Object.assign({}, opts, {headers}));
}

function ddfr(s){const[y,m,d]=s.split('-');return d+'/'+m+'/'+y}

function buildQD(){

async function loadGrid(dv){
  try{
    const ts=Date.now();
    const url=BASE+'?idact=328&idses=S0&CHAMP_SELECTEUR_JOUR='+encodeURIComponent(ddfr(dv)+' '+dayname(dv))+'&_='+ts;
    const r=await pfetchWithSession(url);
    const txt=await r.text();
    log('Grille: '+txt.length+' octets','ld');
    parseGrid(txt);
    return true;
  }catch(e){
    log('Grille indisponible (J+3 pas encore ouverte)','lw');
    return false;
  }
}

function parseGrid(txt){
  S.slotStatus={};
  COURTS.forEach(c=>{S.courtStatus[c.id]={free:true,label:'',maint:false}});
  const lteMap={1:27,2:28,3:29,4:30,5:31,6:32};
  const courtSlotMap={};

  // Parser les selecteurs multi-terrains
  const multiRe=/\$\("([^"]+)"\)[^;]*addClass\("([^"]+)"\)/g;
  let m;
  while((m=multiRe.exec(txt))!==null){
    const sel=m[1],cls=m[2];
    sel.split(',').forEach(part=>{
      const pm=part.trim().match(/#lte(\d+)[^#]*#ch_(\d{2})_(\d{2})/);
      if(pm){
        const lte=parseInt(pm[1]),slot=pm[2]+':'+pm[3];
        if(!courtSlotMap[lte]) courtSlotMap[lte]={};
        if(!courtSlotMap[lte][slot]) courtSlotMap[lte][slot]=cls;
      }
    });
  }

  // Detecter maintenance
  const maintRe=/CLUB\s*-\s*(\d+)[^<]*(?:<[^>]*>)*[^<]*MAINTENANCE/gi;
  while((m=maintRe.exec(txt))!==null){
    const n=parseInt(m[1]);
    const c=COURTS.find(x=>x.n===n);
    if(c) S.courtStatus[c.id].maint=true;
  }

  // Agregation par creneau
  SLOTS_1H.forEach(slot=>{
    let blocked=0,free=0;
    COURTS.forEach(c=>{
      const lte=Object.keys(lteMap).find(k=>parseInt(lteMap[k])===c.id);
      if(!lte) return;
      const cls=(courtSlotMap[parseInt(lte)]||{})[slot]||'';
      /mpleine|fpleine/.test(cls) ? blocked++ : free++;
    });
    S.slotStatus[slot]={free:free>0,allBlocked:free===0,partial:blocked>0&&free>0};
  });
}

function buildCourts(){
  const g=document.getElementById('courtsgrid');
  g.innerHTML='';
  const msg=document.getElementById('gridmsg');
  if(!S.connected){msg.style.display='block';msg.textContent='Connectez-vous d\'abord.';return;}
  if(!S.slot){msg.style.display='block';msg.textContent='Selectionnez un creneau d\'abord.';return;}
  msg.style.display='none';
  COURTS.forEach(c=>{
    const st=S.courtStatus[c.id]||{free:true,maint:false};
    const pi=S.prio.indexOf(c.id);
    let cls='ct';
    if(st.maint) cls+=' maint';
    else if(!st.free) cls+=' na';
    else if(pi===0) cls+=' p1';
    else if(pi>0) cls+=' p2';
    const el=document.createElement('div');
    el.className=cls;
    let badge='';
    if(st.maint) badge='<span class="ct-badge bm">MAINT</span>';
    else if(!st.free) badge='<span class="ct-badge bna">PRIS</span>';
    else if(pi===0) badge='<span class="ct-badge bp1">P1</span>';
    else if(pi>0) badge='<span class="ct-badge bp2">P'+(pi+1)+'</span>';
    el.innerHTML=badge+'<div class="ct-n">'+c.n+'</div><div class="ct-sub">'+(st.label||'Court')+'</div>';
    if(!st.maint&&st.free) el.onclick=()=>cycleCourt(c.id);
    g.appendChild(el);
  });
}

function cycleCourt(id){
  const i=S.prio.indexOf(id);
  if(i===-1) S.prio=[id,...S.prio];
  else if(i===0&&S.prio.length>1) S.prio=[...S.prio.slice(1),id];
  else S.prio=S.prio.filter(x=>x!==id);
  buildCourts(); updateSum();
}

async function startFire(){
  if(S.firing) return;
  S.firing=true;
  setBtn('firing','FEU EN COURS...');
  setDot('firing','Rafale active');
  log('DECLENCHEMENT - Re-login...','lw');

  const ok=await doLogin(true);
  if(!ok){log('Echec re-login.','ler');endFire(false);return;}
  log('Connecte ! Rafale...','lok');

  const rps=parseInt(document.getElementById('brate').value);
  const dur=parseInt(document.getElementById('bdur').value)*1000;
  const stop=document.getElementById('stopok').value==='1';
  const ivl=Math.max(50,Math.floor(1000/rps));
  const t0=Date.now();
  let ap=[...S.prio];
  if(!ap.length) ap=COURTS.filter(c=>S.courtStatus[c.id]?.free!==false&&!S.courtStatus[c.id]?.maint).map(c=>c.id);
  if(!ap.length) ap=COURTS.map(c=>c.id);

  S.iv=setInterval(async()=>{
    if(S.success&&stop){endFire(true);return;}
    if(Date.now()-t0>dur){endFire(false);return;}
    S.attempts++;
    document.getElementById('attn').textContent=S.attempts;
    const cid=ap[0];
    const r=await sendResa(cid);
    if(r.ok){
      S.success=true;
      log('RESERVE ! Court '+(COURTS.find(c=>c.id===cid)?.n)+'  #'+S.attempts,'lok');
      if(PARTNERS[S.partner].val) await confPartner();
      await confFinal();
      if(stop) endFire(true);
    } else {
      if(r.pris&&ap.length>1){
        log('Court '+(COURTS.find(c=>c.id===ap[0])?.n)+' pris > rotation','lw');
        ap.shift();
      }
      if(S.attempts<=5||S.attempts%25===0) log('#'+S.attempts+' '+r.err,'ler');
    }
  },ivl);
}

function endFire(ok){
  clearInterval(S.iv); S.firing=false; S.armed=false;
  document.getElementById('btndis').classList.remove('on');
  if(ok){
    setBtn('ok','RESERVE ! '+S.attempts+' tentative(s)');
    setDot('ok','Confirme ('+S.attempts+')');
  } else {
    setBtn('','ARMER LE SNIPER');
    setDot('ok', S.success?'Reserve':'Echec '+S.attempts+' tentatives');
    if(!S.success) log('Echec apres '+S.attempts+' tentatives.','ler');
  }
}

async function doLogin(silent){
  const lg=document.getElementById('login').value.trim();
  const pw=document.getElementById('pwd').value.trim();
  try {
    // 1. Init session club
    const r1=await pfetch(START+'?club='+CLUB);
    // Recuperer cookie depuis headers si disponible
    const setCookie=r1.headers.get('set-cookie')||'';
    const sessMatch=setCookie.match(/PHPSESSID=([^;]+)/);
    if(sessMatch) SESSION_COOKIE='PHPSESSID='+sessMatch[1];

    // 2. POST club
    await pfetchWithSession(BASE, {
      method:'POST',
      headers:{'Content-Type':'application/x-www-form-urlencoded'},
      body:'club='+CLUB
    });

    // 3. GET page login
    const r3=await pfetchWithSession(BASE);
    const page=await r3.text();

    // Recuperer cookie si pas encore
    if(!SESSION_COOKIE){
      const sc2=r3.headers.get('set-cookie')||'';
      const sm2=sc2.match(/PHPSESSID=([^;]+)/);
      if(sm2) SESSION_COOKIE='PHPSESSID='+sm2[1];
    }

    // Extraire champs obfusques
    const excl=new Set(['idact','idpge','idses','usermd5','idgfcmiid','largeur_ecran','hauteur_ecran','pingmax','pingmin','userid','userkey']);
    const fields=[...page.matchAll(/name="([a-z]{14,24})"/gi)].map(m=>m[1]).filter(f=>!excl.has(f));
    const hf=fields[0]||'msefyrseikuavettedia';
    const lf=fields[1]||'rpydihueaypsad';
    const pf=fields[2]||'fyksnurseouetjar';
    const hash=md5(pw.toUpperCase()+lg.toUpperCase());
    const idpge=(page.match(/name="idpge"\s+value="([^"]+)"/)||['','101-'+Date.now()])[1];

    if(!silent) log('Hash: '+hash.slice(0,8)+'... Champs: '+hf.slice(0,6)+'...','ld');

    const body='idact=101&idpge='+idpge+'&usermd5=&'+hf+'='+hash+'&idgfcmiid=0&largeur_ecran=390&hauteur_ecran=844&pingmax=16&pingmin=16&userid=&userkey=&'+lf+'='+encodeURIComponent(lg)+'&'+pf+'=';

    const r4=await pfetchWithSession(BASE, {
      method:'POST',
      headers:{'Content-Type':'application/x-www-form-urlencoded'},
      body
    });
    const txt=await r4.text();

    // Recuperer nouveau cookie apres login
    const sc4=r4.headers.get('set-cookie')||'';
    const sm4=sc4.match(/PHPSESSID=([^;]+)/);
    if(sm4) SESSION_COOKIE='PHPSESSID='+sm4[1];

    if(SESSION_COOKIE&&!silent) log('Session: '+SESSION_COOKIE.slice(0,20)+'...','ld');

    const ok=txt.includes('idact=328')||txt.includes('CHAMP_SELECTEUR')||
              txt.includes('reservation')||(!txt.includes('mot de passe')&&txt.length>500);

    if(!silent&&ok) log('Login reussi !','lok');
    if(!silent&&!ok) log('Reponse inattendue ('+txt.length+' octets)','lw');
    return ok;
  } catch(e){
    if(!silent) log('Erreur: '+e.message.slice(0,60),'ler');
    return false;
  }
}

async function sendResa(cid){
  try{
    const dv=document.getElementById('tdate').value;
    const cj=encodeURIComponent(ddfr(dv)+' '+dayname(dv));
    const[h,m]=S.slot.split(':').map(Number);

    const rand=Math.floor(Math.random()*99999999999999);
    const idpar=PARTNERS[S.partner].val?PARTNERS[S.partner].val.replace('-',''):'';
    const body='idact=336&idpge=210-'+rand+'&IDOBJ='+sc+'_0_'+cid+'&idses=S0&idcrt='+cid+
      '&idpro=&idpar='+idpar+'&pw=60&dj=2&userid=&usermd5=&club=&B_MOJJO=0'+
      '&LISTE_RESA_BOURSE_DATE_JEU=&LISTE_RESA_BOURSE_HEURE_JEU=&LISTE_RESA_BOURSE_COURT_JEU='+
      '&CHAMP_SELECTEUR_JEU=1&ID_TABLEAU=1%7C'+CLUB+'%7C1&CHAMP_SELECTEUR_JOUR='+cj+'&nc=60';
    const r=await pfetchWithSession(BASE,{
      method:'POST',
      headers:{'Content-Type':'application/x-www-form-urlencoded'},
      body
    });
    const txt=await r.text();
    if(/confirm|valid|succes/i.test(txt)||txt.includes('idact=332')||txt.includes('partenaire')) return{ok:true};
    if(/indisponible|occup|pris|taken/i.test(txt)) return{ok:false,pris:true,err:'Court pris'};
    return{ok:false,err:'HTTP '+r.status};
  }catch(e){return{ok:false,err:e.message.slice(0,30)};}
}
